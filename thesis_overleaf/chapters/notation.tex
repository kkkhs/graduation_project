\begin{notation}

  \begin{notationlist}{2em}
    \item[AP50] IoU 阈值为 0.5 时的平均精度
    \item[AP50-95] 多个 IoU 阈值区间上的平均精度
    \item[IoU] 预测框与真实框的交并比
    \item[Precision] 检测结果中的查准率
    \item[Recall] 检测结果中的查全率
    \item[F1] Precision 与 Recall 的调和平均值
    \item[TP] 真正例数量
    \item[FP] 假正例数量
    \item[FN] 假负例数量
    \item[FPS] 每秒处理图像帧数
    \item[WBF] Weighted Boxes Fusion，加权框融合
    \item[DRENet] 退化重建增强微小船舶检测网络
    \item[FCOS] Fully Convolutional One-Stage Detector
    \item[YOLO] You Only Look Once 系列检测器
  \end{notationlist}

\end{notation}
